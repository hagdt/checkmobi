<script>
    const KEY_VALID = "check1";

    function login() {
        let input = document.getElementById('key').value;
        if(input === KEY_VALID) {
            // Âm thanh thông báo
            let audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            let osc = audioCtx.createOscillator();
            osc.type = 'square';
            osc.frequency.setValueAtTime(440, audioCtx.currentTime);
            osc.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + 0.15);

            document.getElementById('scanner').style.display = 'block';
            
            setTimeout(() => {
                document.getElementById('loginScreen').classList.remove('active');
                document.getElementById('dashScreen').classList.add('active');
                document.getElementById('scanner').style.display = 'none';
                loadAllData();
            }, 1800);
        } else {
            alert("KEY KHÔNG CHÍNH XÁC!");
        }
    }

    async function loadAllData() {
        // 1. Thông tin máy
        const ua = navigator.userAgent;
        document.getElementById('d_name').innerText = getDeviceName(ua);
        document.getElementById('d_vendor').innerText = navigator.vendor || "Hệ thống ẩn";
        document.getElementById('d_os').innerText = navigator.platform;
        document.getElementById('d_lang').innerText = navigator.language.toUpperCase();
        
        // 2. Phần cứng
        document.getElementById('d_cpu').innerText = (navigator.hardwareConcurrency || "8") + " Nhân (Luồng)";
        document.getElementById('d_ram').innerText = (navigator.deviceMemory || "4") + " GB (Ước tính)";
        document.getElementById('d_screen').innerText = window.screen.width + "x" + window.screen.height;
        document.getElementById('d_touch').innerText = navigator.maxTouchPoints + " Điểm chạm";

        // 3. Pin
        if (navigator.getBattery) {
            const battery = await navigator.getBattery();
            const updateBattery = () => {
                document.getElementById('d_bat').innerText = Math.round(battery.level * 100) + "%";
                document.getElementById('d_charge').innerText = battery.charging ? "ĐANG SẠC ⚡" : "KHÔNG";
            };
            updateBattery();
            battery.onlevelchange = updateBattery;
            battery.onchargingchange = updateBattery;
        } else {
            document.getElementById('d_bat').innerText = "Không hỗ trợ";
        }

        // 4. Mạng & IP
        if (navigator.connection) {
            document.getElementById('d_net').innerText = navigator.connection.effectiveType.toUpperCase();
        } else {
            document.getElementById('d_net').innerText = "WIFI/LTE";
        }
        
        try {
            const res = await fetch('https://ipapi.co/json/');
            const data = await res.json();
            document.getElementById('d_ip').innerText = data.ip;
            document.getElementById('d_isp').innerText = data.org;
            document.getElementById('d_loc').innerText = data.city + ", " + data.country_name;
        } catch(e) {
            document.getElementById('d_ip').innerText = "Đang ẩn IP";
        }

        // 5. Cảm biến chuyển động
        if (window.DeviceMotionEvent) {
            window.addEventListener('devicemotion', (event) => {
                if(Math.abs(event.acceleration.x) > 1.5) {
                    document.getElementById('d_motion').innerText = "ĐANG DI CHUYỂN";
                    document.getElementById('d_motion').style.color = "var(--warning)";
                } else {
                    document.getElementById('d_motion').innerText = "ỔN ĐỊNH";
                    document.getElementById('d_motion').style.color = "var(--success)";
                }
            });
        }
    }

    function getDeviceName(ua) {
        if (ua.includes("Samsung")) return "Samsung Galaxy";
        if (ua.includes("iPhone")) return "Apple iPhone";
        if (ua.includes("Pixel")) return "Google Pixel";
        if (ua.includes("Oppo")) return "Oppo Smartphone";
        if (ua.includes("Xiaomi")) return "Xiaomi Mi/Redmi";
        if (ua.includes("Vivo")) return "Vivo Smartphone";
        return "Thiết bị Smartphone";
    }
